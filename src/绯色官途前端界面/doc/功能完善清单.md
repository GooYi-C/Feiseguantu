```vue
      <i class="fas fa-upload"></i> 导入 JSON

<script setup>
function exportData() {
  const json = JSON.stringify(gameStore.data, null, 2);
  const blob = new Blob([json], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `绯色官途_${new Date().toISOString().slice(0, 10)}.json`;
  a.click();
  URL.revokeObjectURL(url);
}

async function handleImport(json: string, selectedSections?: string[]) {
  try {
    const parsed = JSON.parse(json);
    const validated = GameSchema.parse(parsed);
    
    if (selectedSections?.length) {
      // 部分导入
      for (const section of selectedSections) {
        gameStore.updateField(section, validated[section]);
      }
    } else {
      // 全量导入
      await gameStore.saveData(validated);
    }
    
    toastr.success('导入成功');
  } catch (e) {
    toastr.error('导入失败: ' + e.message);
  }
}
</script>
```

#### 2.2.3 关系网优化

**当前问题**:

- 人物多时节点重叠
- 交互有限

**改进方案**:

```typescript
// 动态计算布局
const layout = computed(() => {
  const count = Object.keys(人物库.value).length;
  
  // 根据人数选择布局策略
  if (count <= 12) {
    return { type: 'ring', radius: 200 };
  } else if (count <= 30) {
    return { type: 'ring', radius: 280, layers: 2 };
  } else {
    return { type: 'grid', cols: Math.ceil(Math.sqrt(count)) };
  }
});

// 分层环形布局
function calculateLayeredRingPositions(chars, layers) {
  const positions = [];
  const perLayer = Math.ceil(chars.length / layers);
  
  for (let i = 0; i < chars.length; i++) {
    const layer = Math.floor(i / perLayer);
    const indexInLayer = i % perLayer;
    const countInLayer = Math.min(perLayer, chars.length - layer * perLayer);
    
    const radius = 150 + layer * 100;
    const angle = (2 * Math.PI * indexInLayer) / countInLayer - Math.PI / 2;
    
    positions.push({
      x: centerX + radius * Math.cos(angle),
      y: centerY + radius * Math.sin(angle),
    });
  }
  
  return positions;
}
```

```typescript
import { onMounted, onUnmounted } from 'vue';

export function useKeyboardShortcuts() {
  function handleKeydown(e: KeyboardEvent) {
    // Ctrl/Cmd + S: 保存
    if ((e.ctrlKey || e.metaKey) && e.key === 's') {
      e.preventDefault();
      gameStore.saveData();
    }
    
    // Ctrl/Cmd + F: 聚焦搜索框
    if ((e.ctrlKey || e.metaKey) && e.key === 'f') {
      e.preventDefault();
      document.querySelector<HTMLInputElement>('.search-box input')?.focus();
    }
    
    // Esc: 关闭抽屉/弹窗
    if (e.key === 'Escape') {
      closeCurrentDrawer();
    }
  }
  
  onMounted(() => {
    document.addEventListener('keydown', handleKeydown);
  });
  
  onUnmounted(() => {
    document.removeEventListener('keydown', handleKeydown);
  });
}
```

```typescript
  const history = ref<T[]>([]);
  const historyIndex = ref(-1);
  
  function pushState(state: T) {
    // 如果当前不在最新，删除后续记录
    if (historyIndex.value < history.value.length - 1) {
      history.value = history.value.slice(0, historyIndex.value + 1);
    }

    historyIndex.value = history.value.length - 1;
    
    // 限制历史数量
    if (history.value.length > MAX_HISTORY) {
      history.value.shift();
      historyIndex.value--;
    }
  }
  
  function undo(): T | null {
    if (historyIndex.value > 0) {
      historyIndex.value--;

    }
    return null;
  }
  
  function redo(): T | null {

      historyIndex.value++;

    }
    return null;
  }
```
