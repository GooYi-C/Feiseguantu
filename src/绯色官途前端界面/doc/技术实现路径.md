#### 3.1.2 useGameData.ts 实现

```typescript
// stores/useGameData.ts
import { defineStore } from 'pinia';
import { ref, computed, watch } from 'vue';
import { klona } from 'klona';
import { GameSchema, type GameData } from './schema';

type DataSource = 'mvu' | 'character' | 'chat';

export const useGameData = defineStore('gameData', () => {
  // ═══ State ═══
  const rawData = ref<GameData>(GameSchema.parse({}));
  const loading = ref(false);
  const initialized = ref(false);
  const isDirty = ref(false);
  const lastError = ref<string | null>(null);
  const dataSource = ref<DataSource>('character');

  // ═══ 数据加载 ═══
  async function loadData() {
    loading.value = true;
    lastError.value = null;

    try {
      const result = await loadFromBestSource();
      rawData.value = GameSchema.parse(result.data);
      dataSource.value = result.source;
      initialized.value = true;
      isDirty.value = false;
      console.info(`[绯色官途] 从 ${result.source} 加载数据`);
    } catch (err) {
      lastError.value = err instanceof Error ? err.message : String(err);
      console.error('[绯色官途] 数据加载失败', err);
      toastr.error('数据加载失败');
    } finally {
      loading.value = false;
    }
  }

  async function loadFromBestSource(): Promise<{ data: unknown; source: DataSource }> {
    // 1. 优先 MVU
    try {
      await waitGlobalInitialized('Mvu');
      const mvuData = Mvu.getMvuData({ type: 'message', message_id: 'latest' });
      if (mvuData?.stat_data && Object.keys(mvuData.stat_data).length > 0) {
        return { data: mvuData.stat_data, source: 'mvu' };
      }
    } catch {
      // MVU 不可用，继续尝试
    }

    // 2. 角色卡变量
    try {
      const charVars = getVariables({ type: 'character' });
      if (charVars && Object.keys(charVars).length > 0) {
        return { data: charVars, source: 'character' };
      }
    } catch {
      // 继续尝试
    }

    // 3. 聊天变量
    const chatVars = getVariables({ type: 'chat' });
    return { data: chatVars || {}, source: 'chat' };
  }

  // ═══ 数据保存 ═══
  async function saveData(partial?: Partial<GameData>) {
    loading.value = true;
    try {
      const toSave = partial 
        ? { ...klona(rawData.value), ...partial } 
        : klona(rawData.value);

      await saveToSource(toSave, dataSource.value);

      if (partial) {
        Object.assign(rawData.value, partial);
      }
      isDirty.value = false;
      toastr.success('保存成功');
    } catch (err) {
      lastError.value = err instanceof Error ? err.message : String(err);
      toastr.error('保存失败');
    } finally {
      loading.value = false;
    }
  }

  async function saveToSource(data: GameData, source: DataSource) {
    switch (source) {
      case 'mvu':
        await waitGlobalInitialized('Mvu');
        const mvuData = Mvu.getMvuData({ type: 'message', message_id: 'latest' });
        mvuData.stat_data = data;
        await Mvu.replaceMvuData(mvuData, { type: 'message', message_id: 'latest' });
        break;
      case 'character':
        replaceVariables(data, { type: 'character' });
        break;
      case 'chat':
        replaceVariables(data, { type: 'chat' });
        break;
    }
  }

  // ═══ 分区保存 ═══
  async function saveSection<K extends keyof GameData>(key: K) {
    await saveData({ [key]: rawData.value[key] } as Partial<GameData>);
  }

  // ═══ 字段更新 ═══
  function updateField(path: string, value: unknown) {
    _.set(rawData.value, path, value);
    isDirty.value = true;
  }

  // ═══ 事件监听 ═══
  function setupEventListeners() {
    eventOn(tavern_events.CHAT_CHANGED, () => {
      console.info('[绯色官途] 聊天变化，重新加载');
      loadData();
    });
  }

  return {
    rawData,
    loading,
    initialized,
    isDirty,
    lastError,
    dataSource,
    loadData,
    saveData,
    saveSection,
    updateField,
    setupEventListeners,
  };
});
```

#### 3.1.3 useCharacters.ts 实现

```typescript
// stores/useCharacters.ts
import { defineStore, storeToRefs } from 'pinia';
import { computed } from 'vue';
import { useGameData } from './useGameData';
import { 人物Schema, type 人物 } from './schema';

export const useCharacters = defineStore('characters', () => {
  const gameData = useGameData();
  const { rawData, isDirty } = storeToRefs(gameData);

  // ═══ Getters ═══
  const 人物库 = computed(() => rawData.value.人物库);
  const 人物列表 = computed(() => Object.entries(人物库.value));
  const 人物总数 = computed(() => 人物列表.value.length);

  // 按类型分组
  const 绯色对象列表 = computed(() =>
    人物列表.value
      .filter(([, p]) => p.绯色关系 || p.角色标签?.includes('绯色对象'))
      .map(([name, p]) => ({ name, ...p }))
  );

  const 靠山列表 = computed(() =>
    人物列表.value
      .filter(([, p]) => p.靠山关系 || p.角色标签?.includes('靠山'))
      .map(([name, p]) => ({ name, ...p }))
  );

  const 竞争对手列表 = computed(() =>
    人物列表.value
      .filter(([, p]) => p.竞争关系 || p.角色标签?.includes('竞争对手'))
      .map(([name, p]) => ({ name, ...p }))
  );

  const 家属列表 = computed(() =>
    人物列表.value
      .filter(([, p]) => p.家庭关系 || p.角色标签?.includes('家属'))
      .map(([name, p]) => ({ name, ...p }))
  );

  // ═══ Actions ═══
  function addCharacter(name: string, data: Partial<人物> = {}) {
    if (人物库.value[name]) {
      toastr.warning(`人物「${name}」已存在`);
      return false;
    }
    rawData.value.人物库[name] = 人物Schema.parse({ 姓名: name, ...data });
    isDirty.value = true;
    return true;
  }

  function updateCharacter(name: string, updates: Partial<人物>) {
    if (!人物库.value[name]) {
      toastr.warning(`人物「${name}」不存在`);
      return false;
    }
    Object.assign(rawData.value.人物库[name], updates);
    isDirty.value = true;
    return true;
  }

  function deleteCharacter(name: string) {
    if (!人物库.value[name]) return false;
    delete rawData.value.人物库[name];
    isDirty.value = true;
    return true;
  }

  function getCharacter(name: string): 人物 | null {
    return 人物库.value[name] || null;
  }

  return {
    人物库,
    人物列表,
    人物总数,
    绯色对象列表,
    靠山列表,
    竞争对手列表,
    家属列表,
    addCharacter,
    updateCharacter,
    deleteCharacter,
    getCharacter,
  };
});
```

#### 3.1.4 useLocalCache.ts 实现

```typescript
// stores/useLocalCache.ts
import { defineStore } from 'pinia';
import { ref } from 'vue';

const AVATAR_CACHE_KEY = 'scarlet_avatars';
const MAX_AVATAR_SIZE = 4 * 1024 * 1024; // 4MB

export const useLocalCache = defineStore('localCache', () => {
  const avatarCache = ref<Record<string, string>>({});

  // 初始化加载
  function loadCache() {
    try {
      const cached = localStorage.getItem(AVATAR_CACHE_KEY);
      if (cached) {
        avatarCache.value = JSON.parse(cached);
      }
    } catch {
      avatarCache.value = {};
    }
  }

  // 保存缓存
  function saveCache() {
    try {
      localStorage.setItem(AVATAR_CACHE_KEY, JSON.stringify(avatarCache.value));
    } catch (e) {
      console.warn('[LocalCache] 保存失败，可能超出容量', e);
      // 尝试清理旧数据
      pruneOldAvatars();
    }
  }

  // 获取头像
  function getAvatar(name: string): string | null {
    return avatarCache.value[name] || null;
  }

  // 设置头像
  function setAvatar(name: string, dataUrl: string): boolean {
    if (dataUrl.length > MAX_AVATAR_SIZE) {
      toastr.warning('图片太大，请压缩后上传');
      return false;
    }
    avatarCache.value[name] = dataUrl;
    saveCache();
    return true;
  }

  // 删除头像
  function removeAvatar(name: string) {
    delete avatarCache.value[name];
    saveCache();
  }

  // 清空所有
  function clearAllAvatars() {
    avatarCache.value = {};
    localStorage.removeItem(AVATAR_CACHE_KEY);
    toastr.info('头像缓存已清空');
  }

  // 清理旧数据
  function pruneOldAvatars() {
    const entries = Object.entries(avatarCache.value);
    // 保留最近 10 个
    if (entries.length > 10) {
      avatarCache.value = Object.fromEntries(entries.slice(-10));
      saveCache();
    }
  }

  // 初始化
  loadCache();

  return {
    avatarCache,
    getAvatar,
    setAvatar,
    removeAvatar,
    clearAllAvatars,
  };
});
```

```vue
<template>
  <div class="slider-field">
    <label v-if="label" class="field-label">{{ label }}</label>
    <div class="slider-wrapper">
      <input
        type="range"
        :value="modelValue"
        :min="min"
        :max="max"
        :step="step"
        @input="handleInput"
        class="slider"
      />
      <input
        type="number"
        :value="modelValue"
        :min="min"
        :max="max"
        @input="handleInput"
        class="number-input"
      />
    </div>
    <div class="slider-bar" v-if="showBar">
      <div 
        class="slider-fill" 
        :style="{ width: percentage + '%' }"
        :class="colorClass"
      ></div>
    </div>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue';

const props = withDefaults(defineProps<{
  modelValue: number;
  label?: string;
  min?: number;
  max?: number;
  step?: number;
  showBar?: boolean;
}>(), {
  min: 0,
  max: 100,
  step: 1,
  showBar: true,
});

const emit = defineEmits<{
  (e: 'update:modelValue', value: number): void;
}>();

const percentage = computed(() => 
  ((props.modelValue - props.min) / (props.max - props.min)) * 100
);

const colorClass = computed(() => {
  const pct = percentage.value;
  if (pct >= 80) return 'high';
  if (pct >= 50) return 'mid';
  return 'low';
});

function handleInput(e: Event) {
  const target = e.target as HTMLInputElement;
  const value = Math.min(props.max, Math.max(props.min, Number(target.value)));
  emit('update:modelValue', value);
}
</script>
```

```vue
<template>

      </button>
    </div>

        class="suggestion-btn"
        @click="addSuggestion(s)"

<script setup lang="ts">
import { ref, computed } from 'vue';

const props = withDefaults(defineProps<{
  modelValue: string[];
  placeholder?: string;
  suggestions?: string[];
}>(), {
  placeholder: '添加项目...',
});

const emit = defineEmits<{
  (e: 'update:modelValue', value: string[]): void;
}>();

const newItem = ref('');

const availableSuggestions = computed(() =>
  props.suggestions?.filter(s => !props.modelValue.includes(s)) || []
);

function addItem() {
  const value = newItem.value.trim();
  if (!value || props.modelValue.includes(value)) return;
  
  emit('update:modelValue', [...props.modelValue, value]);
  newItem.value = '';
}

function removeItem(index: number) {
  const newArray = [...props.modelValue];
  newArray.splice(index, 1);
  emit('update:modelValue', newArray);
}

function addSuggestion(s: string) {
  if (!props.modelValue.includes(s)) {
    emit('update:modelValue', [...props.modelValue, s]);
  }
}
</script>
```

```vue
<template>
  <div class="record-table">
    <!-- 表头 -->
    <div class="table-header">
      <span class="col-key">名称</span>
      <span v-for="col in columns" :key="col" class="col">{{ col }}</span>
      <span class="col-actions">操作</span>
    </div>
    
    <!-- 数据行 -->
    <div class="table-body">
      <div
        v-for="[key, item] in paginatedItems"
        :key="key"
        class="table-row"
        :class="{ expanded: expandedKeys.has(key) }"
      >
        <div class="row-summary" @click="toggleRow(key)">
          <span class="col-key">{{ key }}</span>
          <span v-for="col in columns" :key="col" class="col">
            {{ getDisplayValue(item, col) }}
          </span>
          <span class="col-actions">
            <button class="action-btn" @click.stop="$emit('edit', key)">
              <i class="fas fa-edit"></i>
            </button>
            <button class="action-btn danger" @click.stop="$emit('delete', key)">
              <i class="fas fa-trash"></i>
            </button>
          </span>
        </div>
        
        <!-- 展开详情 -->
        <div class="row-detail" v-if="expandedKeys.has(key)">
          <slot name="detail" :item="item" :key="key"></slot>
        </div>
      </div>
    </div>
    
    <!-- 分页 -->
    <div class="table-pagination" v-if="totalPages > 1">
      <button :disabled="page === 1" @click="page--">
        <i class="fas fa-chevron-left"></i>
      </button>
      <span>{{ page }} / {{ totalPages }}</span>
      <button :disabled="page === totalPages" @click="page++">
        <i class="fas fa-chevron-right"></i>
      </button>
    </div>
    
    <!-- 添加按钮 -->
    <button class="add-btn" @click="$emit('add')">
      <i class="fas fa-plus"></i> 新增
    </button>
  </div>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue';

const props = defineProps<{
  data: Record<string, any>;
  columns: string[];
  pageSize?: number;
}>();

const emit = defineEmits<{
  (e: 'add'): void;
  (e: 'edit', key: string): void;
  (e: 'delete', key: string): void;
}>();

const page = ref(1);
const expandedKeys = ref(new Set<string>());

const entries = computed(() => Object.entries(props.data));
const totalPages = computed(() => Math.ceil(entries.value.length / (props.pageSize || 10)));
const paginatedItems = computed(() => {
  const size = props.pageSize || 10;
  const start = (page.value - 1) * size;
  return entries.value.slice(start, start + size);
});

function toggleRow(key: string) {
  if (expandedKeys.value.has(key)) {
    expandedKeys.value.delete(key);
  } else {
    expandedKeys.value.add(key);
  }
}

function getDisplayValue(item: any, col: string) {
  const val = item[col];
  if (val === '无' || val === undefined) return '-';
  if (typeof val === 'object') return '...';
  return val;
}
</script>
```
