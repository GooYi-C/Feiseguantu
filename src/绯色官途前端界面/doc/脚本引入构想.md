# 绯色官途 · 脚本引入构想

## 1. 概述

本文档探讨通过引入独立的 Tavern Helper 脚本来增强"绯色官途"前端界面的交互体验。脚本可以实现前端界面（iframe）本身无法完成的功能，如监听酒馆事件、操作酒馆 UI、与 LLM 通信等。

### 适用场景

| 功能类型      | 前端界面能力 | 脚本能力        |
| ------------- | ------------ | --------------- |
| 监听酒馆事件  | ❌ 受限       | ✅ 完整          |
| 操作酒馆 DOM  | ❌ 跨域限制   | ✅ 可操作        |
| 发送 LLM 请求 | ❌ 不可       | ✅ 可调用        |
| 持久化存储    | ✅ 变量       | ✅ 变量+脚本变量 |
| 显示复杂 UI   | ✅ 主要职责   | ⚠️ 简单 UI       |

## 2. 脚本功能构想

### 2.1 变量自动同步脚本

**目的**: 监听 LLM 消息，自动解析并同步变量到前端界面

**场景**:

- LLM 输出包含 `<stat_data>` 代码块时，自动解析并更新变量
- 支持增量更新（delta）和全量替换
- 前端界面自动刷新显示

```typescript
// src/绯色官途脚本/stat_sync.ts
import { reloadOnChatChange } from '@/util/script';

const STAT_BLOCK_REGEX = /```stat_data\n([\s\S]*?)```/;

// 监听消息接收事件
eventOn(tavern_events.MESSAGE_RECEIVED, async (message) => {
  const match = message.content.match(STAT_BLOCK_REGEX);
  if (!match) return;

  try {
    const statData = JSON.parse(match[1]);

    // 获取当前变量
    const current = getVariables({ type: 'message', message_id: message.id });

    // 合并或替换
    const merged = _.merge({}, current, statData);

    // 写回变量
    await replaceVariables(merged, { type: 'message', message_id: message.id });

    console.info('[绯色官途脚本] 变量已同步', {
      messageId: message.id,
      fields: Object.keys(statData)
    });

    // 通知前端界面刷新
    triggerCustomEvent('scarlet_data_updated', { messageId: message.id });

  } catch (e) {
    console.error('[绯色官途脚本] 变量解析失败', e);
  }
});

// 聊天变化时重载脚本
reloadOnChatChange();

$(() => {
  console.info('[绯色官途脚本] 变量自动同步已启动');
});
```

### 2.2 数据变化提醒脚本

**目的**: 在 LLM 输出导致关键数据变化时，弹出醒目提示

**场景**:

- 好感度/信任度大幅变化
- 新增绯色对象或关系阶段变化
- 危机/机遇变化
- 把柄被发现等高危事件

```typescript
// src/绯色官途脚本/data_alert.ts

interface DataChange {
  path: string;
  oldValue: unknown;
  newValue: unknown;
  severity: 'info' | 'warning' | 'danger';
}

// 监控规则
const WATCH_RULES = [
  {
    pattern: /^人物库\..*\.好感度$/,
    threshold: 10,
    severity: (old: number, next: number) =>
      Math.abs(next - old) >= 20 ? 'warning' : 'info',
    message: (name: string, old: number, next: number) =>
      `${name} 好感度 ${old} → ${next} (${next > old ? '↑' : '↓'}${Math.abs(next - old)})`,
  },
  {
    pattern: /^人物库\..*\.绯色关系\.关系阶段$/,
    severity: () => 'warning',
    message: (name: string, old: string, next: string) =>
      `${name} 关系阶段变化: ${old} → ${next}`,
  },
  {
    pattern: /^暗账\.被握把柄\./,
    severity: () => 'danger',
    message: (key: string) => `⚠️ 新增被握把柄: ${key}`,
  },
  {
    pattern: /^机遇与危机\.潜在危机\./,
    severity: () => 'danger',
    message: (key: string, _, data: any) =>
      `🔥 新增危机: ${data?.危机名称 || key}`,
  },
];

// 数据变化检测
function detectChanges(oldData: GameData, newData: GameData): DataChange[] {
  const changes: DataChange[] = [];

  // 深度对比
  function compare(path: string, oldVal: unknown, newVal: unknown) {
    if (_.isEqual(oldVal, newVal)) return;

    for (const rule of WATCH_RULES) {
      if (rule.pattern.test(path)) {
        changes.push({
          path,
          oldValue: oldVal,
          newValue: newVal,
          severity: rule.severity(oldVal as any, newVal as any),
        });
        break;
      }
    }

    // 递归处理对象
    if (_.isObject(newVal) && !_.isArray(newVal)) {
      for (const key of Object.keys(newVal)) {
        compare(`${path}.${key}`, (oldVal as any)?.[key], (newVal as any)[key]);
      }
    }
  }

  compare('', oldData, newData);
  return changes;
}

// 显示变化通知
function showChangeNotification(changes: DataChange[]) {
  if (changes.length === 0) return;

  const dangerChanges = changes.filter(c => c.severity === 'danger');
  const warningChanges = changes.filter(c => c.severity === 'warning');

  if (dangerChanges.length > 0) {
    toastr.error(
      dangerChanges.map(c => formatChangeMessage(c)).join('<br>'),
      '⚠️ 重大变化',
      { timeOut: 8000, escapeHtml: false }
    );
  }

  if (warningChanges.length > 0) {
    toastr.warning(
      warningChanges.map(c => formatChangeMessage(c)).join('<br>'),
      '数据变化',
      { timeOut: 5000, escapeHtml: false }
    );
  }
}

// 存储上一次数据快照用于对比
let lastSnapshot: GameData | null = null;

eventOn(tavern_events.MESSAGE_RECEIVED, async () => {
  // 等待变量同步完成
  await new Promise(r => setTimeout(r, 500));

  const currentData = getVariables({ type: 'message', message_id: 'latest' });

  if (lastSnapshot) {
    const changes = detectChanges(lastSnapshot, currentData);
    showChangeNotification(changes);
  }

  lastSnapshot = klona(currentData);
});
```

### 2.4 LLM 辅助脚本

**目的**: 在特定条件下自动调用 LLM 生成内容

**场景**:

- 每 N 楼自动生成剧情总结
- 检测到矛盾时提示 LLM 修正
- 自动生成缺失的人物背景

```typescript
// src/绯色官途脚本/llm_assistant.ts

const SUMMARY_INTERVAL = 20; // 每20楼生成一次总结

// 统计当前楼层
function getCurrentFloor(): number {
  return SillyTavern.chat?.length || 0;
}

// 生成剧情总结
async function generateSummary() {
  const floor = getCurrentFloor();
  const startFloor = Math.max(0, floor - SUMMARY_INTERVAL);

  // 获取最近消息
  const recentMessages = SillyTavern.chat.slice(startFloor, floor)
    .map(m => `${m.is_user ? '玩家' : 'AI'}: ${m.mes}`)
    .join('\n\n');

  const prompt = `
请基于以下对话内容，生成一份简洁的剧情总结（200字以内），重点关注：
1. 主要事件和转折
2. 重要人物关系变化
3. 政治局势发展
4. 潜在风险或机遇

对话内容：
${recentMessages}
`;

  try {
    // 使用酒馆的 LLM 接口
    const summary = await triggerSlash(`/gen ${prompt}`);

    // 存储总结
    const summaries = getVariables({ type: 'chat', key: 'scarlet_summaries' }) || [];
    summaries.push({
      floor: `${startFloor + 1}-${floor}`,
      content: summary,
      timestamp: new Date().toISOString(),
    });

    replaceVariables(summaries, { type: 'chat', key: 'scarlet_summaries' });

    toastr.success(`第${startFloor + 1}-${floor}楼剧情已总结`);

  } catch (e) {
    console.error('[绯色官途脚本] 总结生成失败', e);
  }
}

// 监听消息数量
let lastFloor = 0;

eventOn(tavern_events.MESSAGE_RECEIVED, () => {
  const currentFloor = getCurrentFloor();

  // 每 SUMMARY_INTERVAL 楼触发一次
  if (currentFloor > 0 &&
      currentFloor % SUMMARY_INTERVAL === 0 &&
      currentFloor !== lastFloor) {
    lastFloor = currentFloor;

    // 延迟执行，避免干扰正常对话
    setTimeout(() => {
      if (confirm(`已达到第${currentFloor}楼，是否生成剧情总结？`)) {
        generateSummary();
      }
    }, 2000);
  }
});
```

### 2.5 数据备份脚本

**目的**: 定期自动备份游戏数据，防止意外丢失

```typescript
// src/绯色官途脚本/auto_backup.ts

const BACKUP_INTERVAL = 30 * 60 * 1000; // 30分钟
const MAX_BACKUPS = 5;

interface Backup {
  timestamp: string;
  floor: number;
  data: GameData;
}

function createBackup() {
  const data = getVariables({ type: 'message', message_id: 'latest' });
  if (!data || Object.keys(data).length === 0) return;

  const backups: Backup[] = JSON.parse(
    localStorage.getItem('scarlet_backups') || '[]'
  );

  // 添加新备份
  backups.unshift({
    timestamp: new Date().toISOString(),
    floor: SillyTavern.chat?.length || 0,
    data: klona(data),
  });

  // 保留最近 N 个
  while (backups.length > MAX_BACKUPS) {
    backups.pop();
  }

  localStorage.setItem('scarlet_backups', JSON.stringify(backups));
  console.info('[绯色官途脚本] 数据已备份', {
    timestamp: backups[0].timestamp,
    backupCount: backups.length
  });
}

// 恢复备份
function restoreBackup(index: number) {
  const backups: Backup[] = JSON.parse(
    localStorage.getItem('scarlet_backups') || '[]'
  );

  if (index < 0 || index >= backups.length) {
    toastr.error('备份不存在');
    return;
  }

  const backup = backups[index];

  if (!confirm(`确定恢复到 ${backup.timestamp} 的备份吗？当前数据将被覆盖。`)) {
    return;
  }

  replaceVariables(backup.data, { type: 'message', message_id: 'latest' });
  toastr.success('备份已恢复');
}

// 定时备份
let backupTimer: number | null = null;

function startAutoBackup() {
  if (backupTimer) clearInterval(backupTimer);

  backupTimer = setInterval(createBackup, BACKUP_INTERVAL);

  // 立即执行一次
  createBackup();
}

// 暴露恢复函数到全局
(window as any).scarletRestoreBackup = restoreBackup;
(window as any).scarletListBackups = () => {
  const backups: Backup[] = JSON.parse(
    localStorage.getItem('scarlet_backups') || '[]'
  );
  console.table(backups.map((b, i) => ({
    index: i,
    time: b.timestamp,
    floor: b.floor,
  })));
};

$(() => {
  startAutoBackup();
  console.info('[绯色官途脚本] 自动备份已启动');
  console.info('使用 scarletListBackups() 查看备份，scarletRestoreBackup(index) 恢复');
});
```
